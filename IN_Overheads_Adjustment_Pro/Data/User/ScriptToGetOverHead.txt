SELECT 

MAX(INVH_INVOICENUMBER) as INVH_INVOICENUMBER,
BR_NAME,
'2022-01-01 00:00:00.000' as INVH_INVOICEDATE,
INVH_CLIENTLASTNAME,
INVH_CLIENTFIRSTNAME,
INVH_PAYNAME,
INVH_BALANCE

FROM DISC_PROD.MATRIXCARE.HIST_STVHC_T_INVOICEHEADER as M
join DISC_PROD.MATRIXCARE.HIST_STVHC_T_BRANCHES as B
on M.INVH_BRANCHID=B.BR_ID


where INVH_INVOICEDATE > '01/01/2022' 
AND INVH_PAYNAME  LIKE ('%Waiver%')
 
 

GROUP BY 
 
BR_NAME,
 
INVH_CLIENTLASTNAME,
INVH_CLIENTFIRSTNAME,
INVH_PAYNAME,
 
INVH_BALANCE


UNION

SELECT 
INVH_INVOICENUMBER,
BR_NAME,
INVH_INVOICEDATE,
INVH_CLIENTLASTNAME,
INVH_CLIENTFIRSTNAME,
INVH_PAYNAME,
INVH_BALANCE
FROM DISC_PROD.MATRIXCARE.HIST_STVHC_T_INVOICEHEADER as M
join DISC_PROD.MATRIXCARE.HIST_STVHC_T_BRANCHES as B
on M.INVH_BRANCHID=B.BR_ID
where INVH_INVOICEDATE > '01/01/2022' 
AND INVH_BALANCE < 0
AND BR_NAME LIKE ('%H_%')
AND BR_NAME LIKE ('%A_%')
GROUP BY 
INVH_INVOICENUMBER,
BR_NAME,
INVH_INVOICEDATE,
INVH_CLIENTLASTNAME,
INVH_CLIENTFIRSTNAME,
INVH_PAYNAME,
 
INVH_BALANCE
